\begin{frame}{Heart Disease Example - Algorithm}
\begin{algorithm}[H]
    \KwData{v (instance), s (SAT solver with hard clauses inserted)} \\
    \KwResult{axp} \\
    $i \gets 0$; $axp \gets copy(v)$; $n \gets length(v)$\; \\
    \While{$(i < n) \land (n > 1)$}{
        $x \gets v[:i] \cup v[(i + 1):]$\; \\
        \If{s.solve(assumptions=x)}{
            $i \gets i + 1$\;
        }
        \Else{
            $axp \gets x$\; \\
            $v \gets v[:i] + v[(i + 1):]$\; \\
            $n \gets n - 1$\; \\
            $i \gets 0$\;
        }
    }
    \textbf{Output:} axp
    \caption{Find AXp}
\end{algorithm}
\end{frame}
